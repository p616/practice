vector<int> climbingLeaderboard(vector<int> scores, vector<int> alice) 
{
    set<int> rankScores(scores.begin(), scores.end());

    vector<int> ret;
    for (int score : alice)
    {
        // insert returns pair<iterator,bool>
        auto rankAlice = rankScores.insert(score);

        // determine Alice's rank and add to return vector
        ret.push_back(distance(rankAlice.first, rankScores.end()));

        // remove from scores set if added
        if (rankAlice.second)
        {
            rankScores.erase(rankAlice.first);
        }
    }

    return ret;
}
