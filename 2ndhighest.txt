public static int alphaBeta(List<int> pile)
    {
        int alpha = -1;
        int beta = -1;
        foreach (var n in pile)
        {
            if (n > alpha)
            {
                beta = alpha;
                alpha = n;
            }
            else if (n > beta && n != alpha)
            {
                beta = n;
            }               
        }
        return Math.Max(beta, 0);
    }
